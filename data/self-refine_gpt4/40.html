const md5 = require("js-md5");
let email = "someone@example.com";
// trim and lowercase the email to remove spaces
let address = String(email).trim().toLowerCase();
// md5 hash the final string
let hash = md5(address);
// form the Gravatar URL
let gravatar_url = "https://www.gravatar.com/avatar/" + hash;
console.log(gravatar_url);
